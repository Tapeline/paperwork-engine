{# Reusable macro: renders a TOC as a collapsible tree. Expects a list of Note | TOCSection items. #} {% macro render_toc(items, base_path, depth) %}
    {% for item in items %} {% if item.__class__.__name__ == 'TOCSection' %}
  • {{ render_toc(item.toc, base_path, depth + 1) }}
    • {% else %}
  • {{ item.title }}
    • {% endif %} {% endfor %}
{% endmacro %}
